========================================
Test MyList.print_sorted basic behavior
========================================

>>> MyList = __import__('1-my_list').MyList
>>> ml = MyList()
>>> ml.append(3)
>>> ml.append(1)
>>> ml.append(2)
>>> ml
[3, 1, 2]
>>> ml.print_sorted()
[1, 2, 3]
>>> ml
[3, 1, 2]

========================================
Test empty list
========================================

>>> ml = MyList()
>>> ml
[]
>>> ml.print_sorted()
[]
>>> ml
[]

========================================
Test already sorted list
========================================

>>> ml = MyList([1, 2, 3, 4])
>>> ml
[1, 2, 3, 4]
>>> ml.print_sorted()
[1, 2, 3, 4]
>>> ml
[1, 2, 3, 4]

========================================
Test reverse-sorted list
========================================

>>> ml = MyList([5, 4, 3, 2, 1])

[1, 2, 3, 4, 5]
